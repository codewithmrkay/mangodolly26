import pandas as pd
import numpy as np

np.random.seed(0)

data = {
    "Transaction_ID": range(1, 21),
    "Customer_ID": np.random.randint(1001, 1020, 20),
    "Amount": np.random.randint(100, 5000, 20),
    "Payment_Method": np.random.choice(["Cash", "Card", "UPI"], 20),
    "Items_Count": np.random.randint(1, 10, 20)
}

df = pd.DataFrame(data)

print("Transactions Dataset:\n")
print(df)

print("\nStatistical Summary:\n")
print(df.describe())
