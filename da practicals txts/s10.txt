import pandas as pd
from mlxtend.preprocessing import TransactionEncoder
from mlxtend.frequent_patterns import apriori, association_rules

# Dataset
transactions = [
['Bread','Milk'],
['Bread','Diaper','Beer','Eggs'],
['Milk','Diaper','Beer','Coke'],
['Bread','Milk','Diaper','Beer'],
['Bread','Milk','Diaper','Coke']
]

# Convert categorical data to numeric
te = TransactionEncoder()
df = pd.DataFrame(te.fit_transform(transactions), columns=te.columns_)

print("Numeric Dataset:")
print(df)

# Apriori with min_support = 0.5
freq = apriori(df, min_support=0.5, use_colnames=True)
print("\nFrequent Itemsets:")
print(freq)

# Association rules
rules = association_rules(freq, metric="confidence", min_threshold=0.7)
print("\nAssociation Rules:")
print(rules)

# Repeat with different min_support
freq2 = apriori(df, min_support=0.6, use_colnames=True)
print("\nFrequent Itemsets with min_support=0.6:")
print(freq2)
