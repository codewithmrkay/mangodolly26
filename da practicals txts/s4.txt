import pandas as pd
import matplotlib.pyplot as plt
from sklearn.linear_model import LinearRegression

# Load dataset with encoding
data = pd.read_csv("Fish_HeightWeight.csv", encoding='latin1')

print(data.head())

# Input and Output
X = data[['Ã¿Height']]
y = data['Weight']

model = LinearRegression()
model.fit(X, y)

y_pred = model.predict(X)
print('predictions:',y_pred)
print("Intercept:", model.intercept_)
print("Slope:", model.coef_)

plt.scatter(X, y)
plt.plot(X, y_pred)
plt.xlabel("Height")
plt.ylabel("Weight")
plt.title("Fish Weight Prediction")
plt.show()
