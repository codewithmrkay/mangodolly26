Program 1: Banker's Algorithm - Display Matrices (Menu Driven)

#include <stdio.h>
int main() {
 int n = 5, m = 3;
 int i, j, choice;
 // Each process currently holds these many resources
 int allocation[5][3] = {{0,1,0},{2,0,0},{3,0,2},{2,1,1},{0,0,2}};
 // Maximum resources each process can ever need
 int max[5][3] = {{7,5,3},{3,2,2},{9,0,2},{2,2,2},{4,3,3}};
 int need[5][3]; // need = max - allocation
 int available[3]; // resources free right now
 printf("Enter Available resources (3 values): ");
 for(j = 0; j < m; j++)
 scanf("%d", &available[j]);
 // Calculate how much more each process may need
 for(i = 0; i < n; i++)
 for(j = 0; j < m; j++)
 need[i][j] = max[i][j] - allocation[i][j];
 do {
 printf("\n--- Menu ---\n");
 printf("1.Allocation 2.Max 3.Need 4.Available 5.Exit\n");
 printf("Enter choice: ");
 scanf("%d", &choice);
 switch(choice) {
 case 1:
 printf("\nAllocation Matrix:\n");
 for(i = 0; i < n; i++) {
 printf("P%d: ", i);
 for(j = 0; j < m; j++) printf("%d ", allocation[i][j]);
 printf("\n");
 }
 break;
 case 2:
 printf("\nMax Matrix:\n");
 for(i = 0; i < n; i++) {
 printf("P%d: ", i);
 for(j = 0; j < m; j++) printf("%d ", max[i][j]);
 printf("\n");
 }
 break;
 case 3:
 printf("\nNeed Matrix:\n");
 for(i = 0; i < n; i++) {
 printf("P%d: ", i);
 for(j = 0; j < m; j++) printf("%d ", need[i][j]);
 printf("\n");
 }
 break;
 case 4:
 printf("\nAvailable: ");
 for(j = 0; j < m; j++) printf("%d ", available[j]);
 printf("\n");
 break;
 case 5:
 printf("Exiting.\n");
 break;
 default:
 printf("Invalid choice!\n");
 }
 } while(choice != 5);
 return 0;
}

Program 2: Banker's Algorithm - Safe State Detection

#include <stdio.h>
int main() {
 int i, j, k;
 int processes = 6, resources = 4;
 // Resources currently held by each process
 int alloc[6][4] = {{0,3,2,4},{1,2,0,1},{0,0,0,0},
 {3,3,2,2},{1,4,3,2},{2,4,1,4}};
 // Maximum resources each process may ever need
 int max[6][4] = {{6,5,4,4},{4,4,4,4},{0,0,1,2},
 {3,9,3,4},{2,5,3,3},{4,6,3,4}};
 int avail[4] = {3,4,4,2}; // Currently free resources
 int need[6][4]; // need = max - alloc
 int finish[6] = {0}; // 0 = not done, 1 = completed
 int safeSeq[6]; // safe execution order
 int work[4]; // working copy of available
 for(i = 0; i < processes; i++)
 for(j = 0; j < resources; j++)
 need[i][j] = max[i][j] - alloc[i][j];
 for(i = 0; i < resources; i++)
 work[i] = avail[i];
 int count = 0;
 while(count < processes) {
 int found = 0;
 for(i = 0; i < processes; i++) {
 if(finish[i] == 0) {
 int flag = 1;
 // Check if this process can run with current work
 for(j = 0; j < resources; j++)
 if(need[i][j] > work[j]) { flag = 0; break; }
 if(flag) {
 // Process finishes, releases its resources
 for(k = 0; k < resources; k++)
 work[k] += alloc[i][k];
 safeSeq[count++] = i;
 finish[i] = 1;
 found = 1;
 }
 }
 }
 // No process could run = deadlock = unsafe
 if(found == 0) {
 printf("System is NOT in safe state\n");
 return 0;
 }
 }
 printf("System is in SAFE STATE\nSafe sequence: ");
 for(i = 0; i < processes; i++) printf("P%d ", safeSeq[i]);
 return 0;
}

#include <stdio.h>
#include <string.h>
#define MAX 20
int bit[MAX]; // 0 = block free, 1 = block used
struct file {
 char name[20];
 int start; // starting block number
 int len; // number of blocks used
} dir[MAX];
int main() {
 int n, i, j, choice, f = 0;
 char fname[20];
 printf("Enter number of blocks: ");
 scanf("%d", &n);
 for(i = 0; i < n; i++) bit[i] = 0; // all blocks free initially
 while(1) {
 printf("\n1.Bit Vector 2.Create 3.Directory 4.Delete 5.Exit\n");
 printf("Enter choice: ");
 scanf("%d", &choice);
 switch(choice) {
 case 1:
 printf("\nBit Vector: ");
 for(i = 0; i < n; i++) printf("%d ", bit[i]);
 printf("\n");
 break;
 case 2:
 printf("File name: "); scanf("%s", dir[f].name);
 printf("Start block: "); scanf("%d", &dir[f].start);
 printf("Length: "); scanf("%d", &dir[f].len);
 // Mark all contiguous blocks as used
 for(i = dir[f].start; i < dir[f].start + dir[f].len; i++)
 bit[i] = 1;
 f++;
 break;
 case 3:
 printf("\nFile\tStart\tLength\n");

 printf("%s\t%d\t%d\n", dir[i].name, dir[i].start,
dir[i].len);

 for(i = 0; i < f; i++)
 if(strcmp(fname, dir[i].name) == 0)
 // Free all blocks belonging to this file
 for(j = dir[i].start; j < dir[i].start + dir[i].len;
j++)
 bit[j] = 0;
 break;
 case 5: return 0;

 }

Program 4: File Allocation - Linked Allocation with Delete

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <time.h>
#define MAX 50
int bit[MAX]; // 0 = block free, 1 = block used
int next[MAX]; // next[i] = next block after block i (-1 = end)
struct file {
 char name[20];
 int start; // first block of the file
} dir[MAX];
int main() {
 int n, i, choice, f = 0;
 printf("Enter number of disk blocks: ");
 scanf("%d", &n);
 srand(time(0));
 for(i = 0; i < n; i++) {
 bit[i] = rand() % 2; // randomly pre-allocate some blocks
 next[i] = -1;
 }
 while(1) {
 printf("\n1.Bit Vector 2.Create 3.Directory 4.Delete 5.Exit\n");
 printf("Enter choice: ");
 scanf("%d", &choice);
 switch(choice) {
 case 1:
 printf("\nBit Vector: ");
 for(i = 0; i < n; i++) printf("%d ", bit[i]);
 printf("\n");
 break;
 case 2: {
 int blocks, count = 0, prev = -1;
 printf("File name: "); scanf("%s", dir[f].name);
 printf("Blocks needed: "); scanf("%d", &blocks);
 // Find free blocks randomly and link them
 while(count < blocks) {
 int b = rand() % n;
 if(bit[b] == 0) {
 bit[b] = 1;
 if(prev == -1) dir[f].start = b; // first block
 else next[prev] = b; // link to previous
 prev = b;
 count++;
 }
 }
 next[prev] = -1; // end of chain
 printf("File allocated.\n");
 f++;
 break;
 }
 case 3: {
 int j;
 printf("\nFile\tStart\tChain\n");
 for(i = 0; i < f; i++) {
 printf("%s\t%d\t", dir[i].name, dir[i].start);
 j = dir[i].start;
 while(j != -1) { printf("%d->", j); j = next[j]; }
 printf("NULL\n");
 }
 break;
 }
 case 4: {
 char fname[20];
 int j, found = 0;
 printf("File to delete: "); scanf("%s", fname);
 for(i = 0; i < f; i++) {
 if(strcmp(fname, dir[i].name) == 0) {
 found = 1;
 // Traverse chain and free each block
 j = dir[i].start;
 while(j != -1) {
 int tmp = next[j];
 bit[j] = 0; // mark block free
 next[j] = -1; // remove link
 j = tmp;
 }
 // Remove entry from directory
 for(j = i; j < f - 1; j++) dir[j] = dir[j+1];
 f--;
 printf("File deleted.\n");
 break;
 }
 }
 if(!found) printf("File not found!\n");
 break;
 }
 case 5: return 0;
 default: printf("Invalid choice!\n");
 }
 }
}

Create: name=test blocks=3
Choice 3: test 1 1->6->8->NULL
Delete: test
Choice 3: (empty directory)
Blocks 1, 6, 8 are free again.

Program 5: Disk Scheduling - FCFS Algorithm

#include <stdio.h>
#include <stdlib.h>
int main() {
 int n, i, head, total = 0;
 printf("Enter number of requests: "); scanf("%d", &n);
 int req[n];
 printf("Enter request queue: ");
 for(i = 0; i < n; i++) scanf("%d", &req[i]);
 printf("Enter initial head position: "); scanf("%d", &head);
 printf("\nOrder of servicing:\n");
 // FCFS: serve in the order requests arrived, no sorting
 for(i = 0; i < n; i++) {
 printf("%d ", req[i]);
 total += abs(head - req[i]); // distance moved
 head = req[i]; // head moves here
 }
 printf("\nTotal head movement = %d\n", total);
 return 0;
}

Program 6: Disk Scheduling - C-SCAN Algorithm

#include <stdio.h>
#include <stdlib.h>
int main() {
 int req[20], n, head, disk_size;
 int i, j, temp, total = 0;
 printf("Enter number of requests: "); scanf("%d", &n);
 printf("Enter request queue: ");
 for(i = 0; i < n; i++) scanf("%d", &req[i]);
 printf("Enter initial head position: "); scanf("%d", &head);
 printf("Enter total disk blocks: "); scanf("%d", &disk_size);
 int orig_head = head; // save original position
 // Sort requests in ascending order
 for(i = 0; i < n-1; i++)
 for(j = i+1; j < n; j++)
 if(req[i] > req[j]) { temp=req[i]; req[i]=req[j]; req[j]=temp; }
 printf("\nOrder of servicing:\n");
 // Step 1: Serve all requests to the RIGHT of head
 for(i = 0; i < n; i++) {
 if(req[i] >= orig_head) {
 printf("%d ", req[i]);
 total += abs(head - req[i]);
 head = req[i];
 }
 }
 // Step 2: Move head to the last cylinder
 total += abs(head - (disk_size - 1));
 head = disk_size - 1;
 // Step 3: Jump back to cylinder 0 (circular, no service on return)
 total += disk_size - 1;
 head = 0;
 // Step 4: Serve remaining requests from cylinder 0 upward
 for(i = 0; i < n; i++) {
 if(req[i] < orig_head) {
 printf("%d ", req[i]);
 total += abs(head - req[i]);
 head = req[i];
 }
 }
 printf("\nTotal head movement = %d\n", total);
 return 0;
}

Program 7: MPI - Parallel Sum & Average

#include <stdio.h>
#include <stdlib.h>
#include <mpi.h>
#define ARRAY_SIZE 1000
int main(int argc, char *argv[]) {
 int rank, size, i;
 int data[ARRAY_SIZE];
 int local_sum = 0, sum = 0;
 MPI_Init(&argc, &argv); // start MPI
 MPI_Comm_rank(MPI_COMM_WORLD, &rank); // this process ID
 MPI_Comm_size(MPI_COMM_WORLD, &size); // total processes
 // Only process 0 generates the full array
 if(rank == 0) {
 printf("Generating data...\n");
 for(i = 0; i < ARRAY_SIZE; i++)
 data[i] = rand() % 100;
 }
 // Divide array equally and send one part to each process
 MPI_Scatter(data, ARRAY_SIZE/size, MPI_INT,
 data, ARRAY_SIZE/size, MPI_INT, 0, MPI_COMM_WORLD);
 // Each process computes sum of its own part
 for(i = 0; i < ARRAY_SIZE/size; i++)
 local_sum += data[i];
 // Collect all partial sums into one final sum at process 0
 MPI_Reduce(&local_sum, &sum, 1, MPI_INT, MPI_SUM, 0, MPI_COMM_WORLD);
 if(rank == 0) {
 printf("Sum = %d\n", sum);
 printf("Average = %.2f\n", (float)sum / ARRAY_SIZE);
 }
 MPI_Finalize(); // end MPI
 return 0;
}

Program 8: MPI - Find Minimum & Maximum

#include <mpi.h>
#include <stdio.h>
int main(int argc, char *argv[]) {
 int rank, size;
 int number, min, max;
 MPI_Init(&argc, &argv); // start MPI
 MPI_Comm_rank(MPI_COMM_WORLD, &rank); // this process ID
 MPI_Comm_size(MPI_COMM_WORLD, &size); // total processes
 // Give each process a unique number: P0=10, P1=11, P2=12 ...
 number = rank + 10;
 printf("Process %d has number %d\n", rank, number);
 // Find minimum across all processes, result at process 0
 MPI_Reduce(&number, &min, 1, MPI_INT, MPI_MIN, 0, MPI_COMM_WORLD);
 // Find maximum across all processes, result at process 0
 MPI_Reduce(&number, &max, 1, MPI_INT, MPI_MAX, 0, MPI_COMM_WORLD);
 if(rank == 0) {
 printf("\nMinimum = %d\n", min);
 printf("Maximum = %d\n", max);
 }
 MPI_Finalize(); // end MPI
 return 0;
}

/*
* Program 9: Banker's Algorithm - Resource Request
* Sets: G, L
* Compile: gcc 09_resource_request.c -o resource_request
* Run: ./resource_request
*/
#include <stdio.h>
int main() {
int i, j, k;
int n = 5, m = 4; /* 5 processes, 4 resources A B C D */
/* Snapshot from Set-G / Set-L */
int alloc[5][4] = {{0,0,1,2},{1,0,0,0},{1,3,5,4},{0,6,3,2},{0,0,1,4}};
int max[5][4] = {{0,0,1,2},{1,7,5,0},{2,3,5,6},{0,6,5,2},{0,6,5,6}};
int avail[4] = {1,5,2,0};
int need[5][4], finish[5], safeSeq[5], work[4];
/* Calculate Need */
for(i = 0; i < n; i++)
for(j = 0; j < m; j++)
need[i][j] = max[i][j] - alloc[i][j];
/* Display Need Matrix */
printf("Need Matrix:\n");
printf("%-5s A B C D\n", "");
for(i = 0; i < n; i++) {
printf("P%d: ", i);
for(j = 0; j < m; j++) printf("%-3d", need[i][j]);
printf("\n");
}
/* Safety Algorithm */
for(i = 0; i < n; i++) finish[i] = 0;
for(j = 0; j < m; j++) work[j] = avail[j];
int count = 0;
while(count < n) {
int found = 0;
for(i = 0; i < n; i++) {
if(!finish[i]) {
int ok = 1;
for(j = 0; j < m; j++)
if(need[i][j] > work[j]) { ok = 0; break; }
if(ok) {
for(k = 0; k < m; k++) work[k] += alloc[i][k];
safeSeq[count++] = i;
finish[i] = 1;
found = 1;
}
}
}
if(!found) break;
}
if(count == n) {
printf("\nSystem is in SAFE STATE\nSafe sequence: ");
for(i = 0; i < n; i++) printf("P%d ", safeSeq[i]);
printf("\n");
} else {
printf("\nSystem is NOT in safe state\n");
return 0;
}
/* Resource Request */
int pid, req[4];
printf("\nEnter process number for request (0-%d): ", n-1);
scanf("%d", &pid);
printf("Enter request vector (4 values): ");
for(j = 0; j < m; j++) scanf("%d", &req[j]);
/* Check: request <= need? */
for(j = 0; j < m; j++) {
if(req[j] > need[pid][j]) {
printf("Error: Request exceeds maximum need!\n");
return 0;
}
}
/* Check: request <= available? */
for(j = 0; j < m; j++) {
if(req[j] > avail[j]) {
printf("Process P%d must WAIT. Resources not available.\n", pid);
return 0;
}
}
/* Pretend to allocate */
for(j = 0; j < m; j++) {
avail[j] -= req[j];
alloc[pid][j] += req[j];
need[pid][j] -= req[j];
}
/* Check safety again */
for(i = 0; i < n; i++) finish[i] = 0;
for(j = 0; j < m; j++) work[j] = avail[j];
count = 0;
while(count < n) {
int found = 0;
for(i = 0; i < n; i++) {
if(!finish[i]) {
int ok = 1;
for(j = 0; j < m; j++)
if(need[i][j] > work[j]) { ok = 0; break; }
if(ok) {
for(k = 0; k < m; k++) work[k] += alloc[i][k];
safeSeq[count++] = i;
finish[i] = 1;
found = 1;
}
}
}
if(!found) break;
}
if(count == n) {
printf("Request GRANTED. System still in safe state.\n");
printf("New safe sequence: ");
for(i = 0; i < n; i++) printf("P%d ", safeSeq[i]);
printf("\n");
} else {
printf("Request DENIED. Would cause unsafe state. Rolling back.\n");
for(j = 0; j < m; j++) {
avail[j] += req[j];
alloc[pid][j] -= req[j];
need[pid][j] += req[j];
}
}
return 0;
}

/*
* Program 10: Disk Scheduling - SSTF Algorithm
* Set: C
* Compile: gcc 10_sstf_disk.c -o sstf_disk
* Run: ./sstf_disk
* Sample: n=9, queue: 55 58 39 18 90 160 150 38 184, head=50
* Expected Total = 214
*/
#include <stdio.h>
#include <stdlib.h>
int main() {
int n, i, j, head, total = 0;
printf("Enter number of requests: "); scanf("%d", &n);
int req[n], served[n];
printf("Enter request queue: ");
for(i = 0; i < n; i++) { scanf("%d", &req[i]); served[i] = 0; }
printf("Enter initial head position: "); scanf("%d", &head);
printf("\nOrder of servicing:\n");
/* SSTF: always pick the closest unserved request */
for(i = 0; i < n; i++) {
int minDist = 99999, minIdx = -1;
for(j = 0; j < n; j++) {
if(!served[j]) {
int dist = abs(req[j] - head);
if(dist < minDist) { minDist = dist; minIdx = j; }
}
}
served[minIdx] = 1;
printf("%d ", req[minIdx]);
total += abs(req[minIdx] - head);
head = req[minIdx];
}
printf("\nTotal head movement = %d\n", total);
return 0;
}

Program 11: Disk Scheduling - SCAN Algorithm
Compile: gcc 11_scan_disk.c -o scan_disk

/*
* Program 11: Disk Scheduling - SCAN Algorithm (Elevator)
* Set: J
* Compile: gcc 11_scan_disk.c -o scan_disk
* Run: ./scan_disk
* Sample: n=8, queue: 33 99 142 52 197 79 46 65, head=72, dir=right
*/
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
int main() {
int req[20], n, head, disk_size;
int i, j, temp, total = 0;
char dir[10];
printf("Enter number of requests: "); scanf("%d", &n);
printf("Enter request queue: ");
for(i = 0; i < n; i++) scanf("%d", &req[i]);
printf("Enter initial head position: "); scanf("%d", &head);
printf("Enter total disk blocks: "); scanf("%d", &disk_size);
printf("Enter direction (left/right): "); scanf("%s", dir);
int orig_head = head;
/* Sort requests ascending */
for(i = 0; i < n-1; i++)
for(j = i+1; j < n; j++)
if(req[i] > req[j]) { temp=req[i]; req[i]=req[j]; req[j]=temp; }
printf("\nOrder of servicing:\n");
if(strcmp(dir, "right") == 0) {
/* Go right first, then reverse left */
for(i = 0; i < n; i++) {
if(req[i] >= orig_head) {
printf("%d ", req[i]);
total += abs(head - req[i]);
head = req[i];
}
}
/* SCAN goes to end of disk then reverses */
total += abs(head - (disk_size - 1));
head = disk_size - 1;
for(i = n-1; i >= 0; i--) {
if(req[i] < orig_head) {
printf("%d ", req[i]);
total += abs(head - req[i]);
head = req[i];
}
}
} else {
/* Go left first, then reverse right */
for(i = n-1; i >= 0; i--) {
if(req[i] <= orig_head) {
printf("%d ", req[i]);
total += abs(head - req[i]);
head = req[i];
}
}
/* SCAN goes to cylinder 0 then reverses */
total += abs(head - 0);
head = 0;
for(i = 0; i < n; i++) {
if(req[i] > orig_head) {
printf("%d ", req[i]);
total += abs(head - req[i]);
head = req[i];
}
}
}
printf("\nTotal head movement = %d\n", total);
return 0;
}

printf("Enter initial head position: "); scanf("%d", &head);
printf("Enter direction (left/right): "); scanf("%s", dir);
int orig_head = head;
/* Sort requests ascending */
for(i = 0; i < n-1; i++)
for(j = i+1; j < n; j++)
if(req[i] > req[j]) { temp=req[i]; req[i]=req[j]; req[j]=temp; }
printf("\nOrder of servicing:\n");
if(strcmp(dir, "right") == 0) {
/* Go right first - LOOK: stop at last request, don't go to end */
for(i = 0; i < n; i++) {
if(req[i] >= orig_head) {
printf("%d ", req[i]);
total += abs(head - req[i]);
head = req[i];
}
}
/* Reverse, serve left side */
for(i = n-1; i >= 0; i--) {
if(req[i] < orig_head) {
printf("%d ", req[i]);
total += abs(head - req[i]);
head = req[i];
}
}
} else {
/* Go left first - LOOK: stop at lowest request, don't go to 0 */
for(i = n-1; i >= 0; i--) {
if(req[i] <= orig_head) {
printf("%d ", req[i]);
total += abs(head - req[i]);
head = req[i];
}
}
/* Reverse, serve right side */
for(i = 0; i < n; i++) {
if(req[i] > orig_head) {
printf("%d ", req[i]);
total += abs(head - req[i]);
head = req[i];
}
}
}
printf("\nTotal head movement = %d\n", total);
return 0;
}

/*
* Program 13: MPI - Sum of Even Numbers
* Set: K
* Compile: mpicc 13_mpi_even.c -o mpi_even
* Run: mpirun -np 4 ./mpi_even
*/
#include <stdio.h>
#include <stdlib.h>
#include <mpi.h>
#define ARRAY_SIZE 1000
int main(int argc, char *argv[]) {
int rank, size, i;
int data[ARRAY_SIZE];
int local_even_sum = 0, total_even_sum = 0;
MPI_Init(&argc, &argv); /* start MPI */
MPI_Comm_rank(MPI_COMM_WORLD, &rank); /* this process ID */
MPI_Comm_size(MPI_COMM_WORLD, &size); /* total processes */
/* Only process 0 generates the full array */
if(rank == 0) {
printf("Generating %d random numbers...\n", ARRAY_SIZE);
for(i = 0; i < ARRAY_SIZE; i++)
data[i] = rand() % 1000 + 1;
}
/* Divide array equally and send one part to each process */
MPI_Scatter(data, ARRAY_SIZE/size, MPI_INT,
data, ARRAY_SIZE/size, MPI_INT, 0, MPI_COMM_WORLD);
/* Each process adds up only even numbers in its part */
for(i = 0; i < ARRAY_SIZE/size; i++)
if(data[i] % 2 == 0)
local_even_sum += data[i];
printf("Process %d: local even sum = %d\n", rank, local_even_sum);
/* Collect all partial sums at process 0 */
MPI_Reduce(&local_even_sum, &total_even_sum, 1,
MPI_INT, MPI_SUM, 0, MPI_COMM_WORLD);
if(rank == 0)
printf("\nSum of all even numbers = %d\n", total_even_sum);
MPI_Finalize(); /* end MPI */
return 0;
}
